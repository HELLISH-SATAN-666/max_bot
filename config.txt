from __future__ import annotations

import os
from pathlib import Path


BASE_DIR = Path(__file__).resolve().parent


def _load_bool(name: str, default: bool) -> bool:
    value = os.getenv(name)
    if value is None:
        return default
    return value.strip().casefold() in {"1", "true", "yes", "on"}


def _load_int(name: str, default: int) -> int:
    value = os.getenv(name)
    if value is None:
        return default
    try:
        return int(value.strip())
    except ValueError:
        return default


def _load_admin_user_ids_from_file(path: str | Path) -> list[int]:
    file_path = Path(path)
    if not file_path.exists():
        return []

    normalized = file_path.read_text(encoding="utf-8").replace(",", " ").split()
    result: list[int] = []
    seen: set[int] = set()
    for item in normalized:
        try:
            admin_id = int(item.strip())
        except ValueError:
            continue
        if admin_id in seen:
            continue
        seen.add(admin_id)
        result.append(admin_id)

    return result

ADMIN_USER_IDS_FILE_PATH = os.getenv("ADMIN_USER_IDS_FILE_PATH", str(BASE_DIR / "admins.txt"))
ADMIN_USER_IDS = _load_admin_user_ids_from_file(ADMIN_USER_IDS_FILE_PATH)

LONG_POLL_LIMIT = _load_int("LONG_POLL_LIMIT", 100)

LONG_POLL_RETRY_DELAY_SECONDS = _load_int("LONG_POLL_RETRY_DELAY_SECONDS", 5)

LOG_LEVEL = os.getenv("LOG_LEVEL", "INFO").strip().upper() or "INFO"
